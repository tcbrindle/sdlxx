843  static void run_callback(void* user_data, const char* /* name */,
844  const char* old_value, const char* new_value) {
845  auto self = static_cast<hint_callback*>(user_data);
846  self->func(self->name, old_value, new_value);
847  }
848 
849  public:
850  hint_callback(hint name, Func&& func)
851  : func(std::forward<Func>(func)), name(name) {
852  c_call(::SDL_AddHintCallback, name, run_callback, this);
853  }
854 
855  ~hint_callback() {
856  c_call(::SDL_DelHintCallback, name, run_callback, this);
857  }
858 
859  hint_callback(hint_callback&&) = default;
860  hint_callback& operator=(hint_callback&&) = default;
861 
862  private:
863  Func func;
864  hint name;
865  };

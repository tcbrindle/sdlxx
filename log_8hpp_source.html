225  template <typename Func>
226  struct log_output_cb {
227  static void callback(void* data, int category,
228  ::SDL_LogPriority priority, const char* message) {
229  auto self = static_cast<log_output_cb*>(data);
230  return self->func(category, static_cast<log_priority>(priority),
231  message);
232  }
233 
234  log_output_cb(Func func) : func(func) {
235  // There doesn't seem to be any way to restore the default log
236  // handler
237  // in SDL (setting the log function to nullptr stops logging
238  // completely)
239  // so instead we save the old log function and restore it on
240  // destruction.
241  // This could lead to lifetime issues when multiple log output
242  // callbacks
243  // are used, so... don't do that?
244  ::SDL_LogGetOutputFunction(&old_func, &old_user_data);
245  ::SDL_LogSetOutputFunction(callback, this);
246  }
247 
248  ~log_output_cb() {
249  ::SDL_LogSetOutputFunction(old_func, old_user_data);
250  }
251 
252  // Move-only
253  log_output_cb(const log_output_cb&) = delete;
254 
255  log_output_cb& operator=(const log_output_cb&) = delete;
256 
257  log_output_cb(log_output_cb&&) = default;
258 
259  log_output_cb& operator=(log_output_cb&&) = default;
260 
261  Func func;
262  ::SDL_LogOutputFunction old_func = nullptr;
263  void* old_user_data = nullptr;
264  };
